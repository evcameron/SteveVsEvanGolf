async function fetchOverallDetails() { try { const response = await fetch(DETAILS_URL + '&cb=' + new Date().getTime()); const data = await response.text(); const rows = data.split(/\r?\n/).map(row => row.split(',')); const list = document.getElementById('breakdown-list'); list.innerHTML = ''; let tournamentCount = 0; rows.forEach(col => { const rowYear = col[5]?.replace(/"/g, '').trim(); const label = col[3]?.replace(/"/g, '').trim(); if (!label || label === "Category" || label === "" || rowYear !== CONFIG.WebsiteYear) return; const type = col[0]?.replace(/"/g, '').trim().toLowerCase(); const weight = col[4]?.replace(/"/g, '').trim(); const evPts = col[6]?.replace(/"/g, '').trim(); const stPts = col[7]?.replace(/"/g, '').trim(); const evWins = col[8]?.replace(/"/g, '').trim(); const stWins = col[9]?.replace(/"/g, '').trim(); if (label.toLowerCase() === 'total') { document.getElementById('evan-points').innerText = evPts; document.getElementById('steve-points').innerText = stPts; } else { tournamentCount++; // Check if it's a major to hide win counts in the breakdown rows const isMajor = type.includes('major'); const evWinDisp = isMajor ? '' : `${evWins}w`; const stWinDisp = isMajor ? '' : `${stWins}w`; list.innerHTML += `
${label} ${weight ? `${weight}` : ''}
${evPts} ${stPts} ${evWinDisp} ${stWinDisp}
`; if (tournamentCount === 3) list.innerHTML += `
`; } }); } catch (e) { console.error(e); } }
